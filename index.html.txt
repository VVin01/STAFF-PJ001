  <script>
    const dummyStaff = [
      {
        name: "Daw Nandar",
        id: "CEO-001",
        position: "Chief Executive Officer",
        branch: "Head Office",
        contact: "09-123456789",
        photo: "https://randomuser.me/api/portraits/women/44.jpg",
        join: "2015-01-10",
        duty: "Overall company leadership and strategic planning."
      },
      {
        name: "Ko Zaw Min",
        id: "MGR-002",
        position: "Branch Manager",
        branch: "Yangon",
        contact: "09-987654321",
        photo: "https://randomuser.me/api/portraits/men/35.jpg",
        join: "2018-05-20",
        duty: "Managing Yangon operations and team leadership."
      }
    ];

    const loginBox = document.getElementById('loginBox');
    const dashboard = document.getElementById('dashboard');
    const staffTree = document.getElementById('staffTree');

    function login() {
      const u = document.getElementById('username').value;
      const p = document.getElementById('password').value;
      if (u === "admin" && p === "1234") {
        loginBox.style.display = 'none';
        dashboard.style.display = 'block';
        loadStaff();
      } else {
        alert("Invalid login. Hint: admin / 1234");
      }
    }

    function loadStaff() {
      staffTree.innerHTML = "";
      dummyStaff.forEach((s, i) => {
        const card = document.createElement('div');
        card.className = 'staff-card';
        card.innerHTML = `
          <img src="${s.photo}" />
          <h4>${s.name}</h4>
          <p>${s.position}</p>
        `;
        card.onclick = () => openModal(i);
        staffTree.appendChild(card);
      });
    }

    function openModal(index) {
      const s = dummyStaff[index];
      document.getElementById('modalPhoto').src = s.photo;
      document.getElementById('modalName').textContent = s.name;
      document.getElementById('modalID').textContent = s.id;
      document.getElementById('modalPosition').textContent = s.position;
      document.getElementById('modalBranch').textContent = s.branch;
      document.getElementById('modalContact').textContent = s.contact;
      document.getElementById('modalJoin').textContent = s.join;
      document.getElementById('modalDuty').textContent = s.duty;
      document.getElementById('staffModal').style.display = 'flex';
    }

    function closeModal() {
      document.getElementById('staffModal').style.display = 'none';
    }
  </script>
